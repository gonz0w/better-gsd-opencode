const { CacheEngine } = require('./src/lib/cache.js');

// Benchmark function
function benchmark(cache, iterations = 1000) {
  const times = [];
  for (let i = 0; i < iterations; i++) {
    const key = 'bench_' + (i % 100);
    const start = performance.now();
    cache.set(key, 'value_' + i);
    cache.get(key);
    times.push(performance.now() - start);
  }
  times.sort((a, b) => a - b);
  return {
    p50: times[Math.floor(iterations * 0.5)],
    p95: times[Math.floor(iterations * 0.95)],
    p99: times[Math.floor(iterations * 0.99)],
    avg: times.reduce((a, b) => a + b, 0) / iterations
  };
}

const cache = new CacheEngine();
const results = benchmark(cache);
console.log('Benchmark results (ms):');
console.log('  p50:', results.p50.toFixed(3));
console.log('  p95:', results.p95.toFixed(3));
console.log('  p99:', results.p99.toFixed(3));
console.log('  avg:', results.avg.toFixed(3));
console.log('Statement cache:', cache.status().statementCache ? 'ENABLED' : 'DISABLED');
"</verify>
